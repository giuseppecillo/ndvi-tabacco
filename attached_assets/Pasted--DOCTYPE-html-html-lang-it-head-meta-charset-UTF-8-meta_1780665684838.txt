<script>

function calcola() {

// Estrazione valori dai campi

let resa = parseFloat(document.getElementById('resa').value) || 0;

let azotoTot = parseFloat(document.getElementById('azotoTot').value) || 0;

let giorni = parseFloat(document.getElementById('giorni').value) || 0;


let n1 = parseFloat(document.getElementById('n1').value) || 0;

let n2 = parseFloat(document.getElementById('n2').value) || 0;

let n3 = parseFloat(document.getElementById('n3').value) || 0;

let n4 = parseFloat(document.getElementById('n4').value) || 0;

let n5 = parseFloat(document.getElementById('n5').value) || 0;



// Logica di calcolo Agronomica

let media = (n1 + n2 + n3 + n4 + n5) / 5;

let ottimale = 0.50 + ((giorni - 30) * 0.01);

let discostamento = Math.max(0, ottimale - media);


let dose = (discostamento * 500) * (resa / 4.5);


// Limite di sicurezza (non superare metà del budget in un solo intervento)

let limiteMax = azotoTot / 2;

if (dose > limiteMax) dose = limiteMax;

if (media >= ottimale) dose = 0;



// Aggiornamento interfaccia testuale

document.getElementById('resMedia').innerText = media.toFixed(3);

document.getElementById('resOttimale').innerText = ottimale.toFixed(3);

document.getElementById('resDiff').innerText = discostamento.toFixed(3);

document.getElementById('resDose').innerText = dose.toFixed(1);



return { giorni, resa, n1, n2, n3, n4, n5, media, ottimale, discostamento, dose };

}



function salvaOsservazione() {

let dati = calcola();

let tbody = document.getElementById('corpoTabella');

let tr = document.createElement('tr');


// Inserimento della nuova riga nella tabella

tr.innerHTML = `

<td>${dati.giorni}</td>

<td>${dati.resa}</td>

<td>${dati.n1}</td>

<td>${dati.n2}</td>

<td>${dati.n3}</td>

<td>${dati.n4}</td>

<td>${dati.n5}</td>

<td style="background:#e0f7fa;"><strong>${dati.media.toFixed(3)}</strong></td>

<td>${dati.ottimale.toFixed(3)}</td>

<td>${dati.discostamento.toFixed(3)}</td>

<td style="background:#ffcdd2; color:#c62828;"><strong>${dati.dose.toFixed(1)}</strong></td>

`;

tbody.appendChild(tr);

}



// Esegue il calcolo appena apri la pagina

calcola();

</script>
